@colorDark: #1b1a1a;
// https://stackoverflow.com/questions/29591465/use-css-variables-with-rgba-for-gradient-transparency
@colorDarkRgb: red(@colorDark), green(@colorDark), blue(@colorDark);
@splitLayoutThreshold: 1300px;

.darkVars() {
	--bg-color: @colorDark;
	// https://stackoverflow.com/questions/29591465/use-css-variables-with-rgba-for-gradient-transparency
	--bg-color-rgb: @colorDarkRgb;
	--muted-color: #555555;
	--text-color: #dddddd;
}
.lightVars() {
	--bg-color: white;
	--bg-color-rgb: 255, 255, 255;
	--muted-color: #bbbbbb;
	--text-color: @colorDark;
}

:root {
	--beard-color: black;
	--gutter: 16px;
	.lightVars();

	@media (prefers-color-scheme: dark) {
		.darkVars();
	}
}

/** Inlines sources and minifies the resulting markup (when in prod) */
function compileHtml() {
	const stream = src('./app/index.html')
		.pipe(inlinesource({
			// only minify inline sources in prod
			compress: isProd()
		}));

	// only minify HTML in prod
	if (isProd())
		stream.pipe(htmlmin({
			collapseWhitespace: true
		}));

	stream.pipe(dest('./bld'));
	return stream;
}
